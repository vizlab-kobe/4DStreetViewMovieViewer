38 void Model::load()
39 {
40  std::string dirname = m_input->dirname();
41 
42  kvs::Directory data_dir( dirname ); // inputで得たディレクトリ名からディレクトリクラスのdata_dirを定義
43  if ( !data_dir.exists() ) { throw; }
44  if ( !data_dir.isDirectory() ) { throw; }
45 
46  /*------------------------------------------*/
47  /* dirname以下のファイル構造を調べる */
48  /*------------------------------------------*/
49  /* directoryならパスを取得　(m_directories) */
50  /* fileなら全てのfileを取得 (all_files) 　 */
51  /*------------------------------------------*/
52  kvs::FileList all_files;
53 
54  DIR* dir = opendir( data_dir.path().c_str() ); // ディレクトリをopenする
55  if ( !dir ) { throw; }
56 
57  struct dirent* ent;
58  while ( ( ent = readdir( dir ) ) != NULL )
59  {
60  const std::string path( data_dir.path() ); // パス
61  const std::string name( ent->d_name ); // ファイル名
62  if ( name != "." )
63  {
64  if ( name != ".." )
65  {
66  if ( name != ".DS_Store" )
67  {
68  kvs::Directory all_dir( path + data_dir.Separator() + name );
69  if ( all_dir.isDirectory() ) // ディレクトリであれば
70  {
71  m_directories.push_back( all_dir.path() ); // m_directoriesにパスをpush_backする
72  std::sort( m_directories.begin(), m_directories.end() ); // 昇順にsortする
73  m_is_directory = true;
74  m_is_file = false;
75  }
76  else // ファイルであれば
77  {
78  kvs::File file( path + data_dir.Separator() + name );
79  if ( file.isFile() )
80  {
81  all_files.push_back( file ); // all_filesにfileをpush_backする
82  std::sort( all_files.begin(), all_files.end() ); // 昇順にsortする
83  m_is_directory = false;
84  m_is_file = true;
85  }
86  }
87  }
88  }
89  }
90  }
91  /*-----------------------------------*/
92  /* directory内のファイルを取得する */
93  /*-----------------------------------*/
94  /* m_directories → m_file_lists */
95  /* all_files → m_files */
96  /*-----------------------------------*/
97  if ( m_is_directory )
98  {
99  for ( size_t i = 0; i < m_directories.size(); i++ )
100  {
101  kvs::Directory type_dir( m_directories[i] );
102  if ( !type_dir.exists() ) { throw; }
103  if ( !type_dir.isDirectory() ) { throw; }
104 
105  const std::string ext = m_input->extension();
106  const kvs::FileList& files = type_dir.fileList( true );
107  kvs::FileList file_list;
108  for ( size_t j = 0; j < files.size(); j++ )
109  {
110  if ( files[j].extension() == ext )
111  {
112  file_list.push_back( files[j] ); // 指定の拡張子のファイルのみfile_listへpush_backする
113  }
114  }
115  m_file_lists.push_back( file_list ); // file_listをm_file_listsへpush_backする
116  }
117  }
118 
119  if ( m_is_file )
120  {
121  const std::string ext = m_input->extension();
122  for ( size_t i = 0; i < all_files.size(); i++ )
123  {
124  if ( all_files[i].extension() == ext )
125  {
126  m_files.push_back( all_files[i] ); // 指定の拡張子のファイルのみm_filesへpush_backする
127  }
128  }
129  }
130 
131  // 各変数の値を取得
132  m_camera_position = m_input->position();
133  m_camera_array_dimensions = m_input->dimensions();
134  m_frame_rate = m_input->frame_rate();
135  m_data_info = m_input->dataInfo();
136  m_flip_data = 0;
137 
138  this->setup_object( this->camera_position_index() ); // カメラ位置に相当するモデルをオブジェクトに登録
139 }
