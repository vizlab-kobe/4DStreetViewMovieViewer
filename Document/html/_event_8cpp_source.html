50 * @bug FocusMode実行時、挙動が少しおかしい気がする。要確認。

52 /*==========================================================================*/
53 void Event::focusMode()
54 {
55  m_view->movieScreen().scene()->objectManager()->resetXform();
56 
57  const kvs::Vec3i dimensions = m_model->cameraArrayDimensions();
58  float target_x = ( dimensions.x() - 1.0 ) / 2.0;
59  float target_y = ( dimensions.y() - 1.0 ) / 2.0;
60  float target_z = ( dimensions.z() - 1.0 ) / 2.0;
61 
62  const kvs::Vec3 target = kvs::Vec3 ( target_x, target_y, target_z );
63  const kvs::Vec3 eye = kvs::Vec3( m_model->cameraPosition() );
64  const kvs::Vec3 look_at = target - eye;
65  const kvs::Vec3 up = kvs::Vec3(0.0, 1.0, 0.0);
66  const kvs::Vec3 eye_x = look_at.cross(up);
67  const kvs::Vec3 eye_y = look_at.cross(eye_x);
68  const kvs::Vec3 eye_z = -look_at.normalized();
69  const kvs::Mat3 rot = kvs::Mat3(eye_x.x(), eye_y.x(), eye_z.x(),
70  eye_x.y(), eye_y.y(), eye_z.y(),
71  eye_x.z(), eye_y.z(), eye_z.z()
72  );
73 
74  m_view->movieScreen().scene()->objectManager()->rotate( rot );
75 }

152 /*==========================================================================*/
153 void Event::mouseDoubleClickEvent( kvs::MouseEvent* event )
154 {
155  if ( m_model->isSet() )
156  {
157  const kvs::Vec3 p = kvs::Vec3( 0, 0, 0 );
158  const kvs::Vec3 a = kvs::Vec3( 0, 0, -1 );
159  const kvs::Mat3 R = m_view->movieScreen().scene()->object()->xform().rotation();
160  const kvs::Vec3 dir = ( a - p ) * R.inverted();
161 
162  const int x = kvs::Math::Round( dir.x() );
163  const int y = kvs::Math::Round( dir.y() );
164  const int z = kvs::Math::Round( dir.z() );
165  const kvs::Vec3i& pos = m_model->cameraPosition();
166  kvs::Vec3i d( x, y, z );
167  switch ( event->modifiers() )
168  {
169  case kvs::Key::ShiftModifier: d *= -1; break;
170  default: break;
171  }
172 
173  kvs::Timer timer( kvs::Timer::Start );
174  m_model->updateCameraPosition( pos + d );
175 
176  if ( m_controller->focusModeBox().checkState() == Qt::Checked )
177  {
178  this->focusMode();
179  }
180 
181  timer.stop();
182  local::Program::Logger().pushPositionChangeTime( timer.msec() );
183 
184  m_view->movieScreen().update( m_model );
185  }
186 }

209 void Event::keyPressEvent( kvs::KeyEvent* event )
210 {
211  if ( m_model->isSet() )
212  {
213  const kvs::Vec3i& pos = m_model->cameraPosition();
214  kvs::Vec3i d( 0, 0, 0 );
215  switch ( event->key() )
216  {
217  case kvs::Key::Right: d.x() = +1; break;
218  case kvs::Key::Left: d.x() = -1; break;
219  case kvs::Key::Up: d.y() = +1; break;
220  case kvs::Key::Down: d.y() = -1; break;
221  case kvs::Key::PageUp: d.z() = +1; break;
222  case kvs::Key::PageDown: d.z() = -1; break;
223  default: break;
224  }
225 
226  kvs::Timer timer( kvs::Timer::Start );
227  m_model->updateCameraPosition( pos + d );
228  m_controller->cameraPositionChanged();
229 
230  if ( m_controller->focusModeBox().checkState() == Qt::Checked )
231  {
232  this->focusMode();
233  }
234 
235  timer.stop();
236  local::Program::Logger().pushPositionChangeTime( timer.msec() );
237  m_view->movieScreen().update( m_model );
238 
239  typedef lib4dsv::SphericalMapMovieRenderer Renderer;
240  Renderer* renderer = Renderer::DownCast( m_view->movieScreen().scene()->renderer("Renderer") );
241  switch ( event->key() )
242  {
243  case kvs::Key::Space:
244  {
245  if ( renderer->isEnabledAutoPlay() )
246  {
247  renderer->setEnabledAutoPlay( false );
248  m_controller->autoButton().setText( "Play" );
249  }
250  else
251  {
252  renderer->setEnabledAutoPlay( true );
253  m_controller->autoButton().setText( "Stop" );
254  }
255  m_view->movieScreen().update( m_model );
256  break;
257  }
258 
259  case kvs::Key::a:
260  {
261  const int index = 0;
262  m_controller->slider().setValue( index );
263  renderer->setCurrentFrameIndex( index );
264  renderer->setFrameIndex( index );
265  m_view->movieScreen().update( m_model, index );
266  break;
267  }
268 
269  case kvs::Key::b:
270  {
271  break;
272  }
273 
274  case kvs::Key::c:
275  {
276  const kvs::Vec3i pos = m_model->cameraPosition();
277 
278  size_t number_of_directories = m_model->numberOfDirectories();
279  if ( m_model->flipData() == number_of_directories - 1 )
280  {
281  m_model->setFlipData( 0 );
282  kvs::Directory directory( m_model->directoryPath( m_model->flipData() ) );
283  QString directory_name( QString::fromStdString( directory.name() ) );
284  m_controller->flipDataButton().setText( directory_name );
285  }
286  else
287  {
288  size_t flip_data = m_model->flipData();
289  flip_data++;
290  m_model->setFlipData( flip_data );
291  kvs::Directory directory( m_model->directoryPath( m_model->flipData() ) );
292  QString directory_name( QString::fromStdString( directory.name() ) );
293  m_controller->flipDataButton().setText( directory_name );
294  }
295 
296  m_model->updateCameraPosition( pos );
297  m_view->movieScreen().update( m_model );
298  break;
299  }
300 
301  case kvs::Key::e:
302  {
303  const int index = m_model->objectPointer()->device().numberOfFrames() - 1;
304  m_controller->slider().setValue( index );
305  renderer->setCurrentFrameIndex( index );
306  renderer->setFrameIndex( index );
307  m_view->movieScreen().update( m_model, index );
308  break;
309  }
310 
311  case kvs::Key::i:
312  {
313  }
314 
315  case kvs::Key::l:
316  {
317  const bool state = m_controller->loopBox().isChecked();
318  m_controller->loopBox().setChecked( !state );
319  renderer->setEnabledLoopPlay( !state );
320  m_view->movieScreen().update( m_model );
321  break;
322  }
323 
324  case kvs::Key::r:
325  {
326  const bool state = m_controller->reverseBox().isChecked();
327  m_controller->reverseBox().setChecked( !state );
328  renderer->setEnableReversePlay( !state );
329  m_view->movieScreen().update( m_model );
330  break;
331  }
332 
333  case kvs::Key::s:
334  {
335  int sc_width = m_view->movieScreen().width() + 50;
336  int sc_height = m_view->movieScreen().height() + 50;
337 
338  m_view->movieScreen().resize( sc_width, sc_height );
339  m_view->movieScreen().update( m_model );
340  break;
341  }
342 
343  case kvs::Key::S:

345  int sc_width = m_view->movieScreen().width() - 50;

348  m_view->movieScreen().resize( sc_width, sc_height );
349  m_view->movieScreen().update( m_model );
350  break;
351  }
352 
353  case kvs::Key::t:
354  {
355  typedef lib4dsv::SphericalMapMovieRenderer Renderer;
356  Renderer* renderer = Renderer::DownCast( m_view->movieScreen().scene()->renderer("Renderer") );
357  int index = renderer->frameIndex();
358  m_enable_loop_play = renderer->isEnabledLoopPlay();
359  index++;
360 
361  if ( m_enable_loop_play )
362  {
363  if ( index == m_model->objectPointer()->device().numberOfFrames() )
364  {
365  index = 0;
366  }
367  }
368  else

373  }

377  m_controller->slider().setValue( index ); //sliderに今の位置を反映
378  renderer->setCurrentFrameIndex( index );
379  renderer->setFrameIndex( index );
380  m_view->movieScreen().update( m_model, index );
381  break;
382  }
383 
384  case kvs::Key::T:
385  {
386  typedef lib4dsv::SphericalMapMovieRenderer Renderer;
387  Renderer* renderer = Renderer::DownCast( m_view->movieScreen().scene()->renderer("Renderer") );
388  int index = renderer->frameIndex();
389  m_enable_loop_play = renderer->isEnabledLoopPlay();
390  index--;
391 
392  if ( m_enable_loop_play )

407  m_model->objectPointer()->device().setNextFrameIndex( index );
408  m_controller->slider().setValue( index );
409  renderer->setCurrentFrameIndex( index );
410  renderer->setFrameIndex( index );
411  m_view->movieScreen().update( m_model, index );
412  break;
413  }

415  default: break;
