33  /*==========================================================================*/
34  /**
35  * @brief コンストラクタ
36  * @param model Modelへのポインタ
37  * @param view Viewへのポインタ
38  * @param controller Controllerへのポインタ
39  */
40  /*==========================================================================*/
41  TimerEvent( local::Model* model, local::View* view, local::Controller* controller ):
42  m_model( model ),
43  m_view( view ),
44  m_controller( controller ) {}
45 
46  /*==========================================================================*/
47  /**
48  * @brief タイマーイベントを実行する関数
49  * @param event TimerEventへのポインタ
50  * @details Controllerの更新
51  * @details スクリーンの再描画
52  * @details AutoPlay中でLoopPlayしていない時に最後（最初）のフレームまできたらAutoPlayをOFFにする
53  */
54  /*==========================================================================*/
55  void update( kvs::TimeEvent* event )
56  {
57  if ( m_model->isSet() )
58  {
59  typedef lib4dsv::SphericalMapMovieRenderer Renderer;
60  Renderer* renderer = Renderer::DownCast( m_view->movieScreen().scene()->renderer("Renderer") );
61  const int index = renderer->currentFrameIndex();
62 
63  if ( renderer->isEnabledAutoPlay() ) // AutoPlayの状態
64  {
65  if ( !renderer->isEnabledLoopPlay() ) // LoopPlay Off
66  {
67  if ( !renderer->isEnabledReversePlay() ) // ReversePlay Off
68  {
69  const int nframes = (int)m_model->objectPointer()->device().numberOfFrames();
70  if ( index == nframes - 1 )
71  {
72  renderer->disableAutoPlay(); // 最後のフレームまできたらAutoPlayをOFFにする
73  }
74  }
75  else // ReversePlay On
76  {
77  if ( index == 0 )
78  {
79  renderer->disableAutoPlay(); // 一番初めのフレームまで戻ったらAutoPlayをOFFにする
80  }
81  }
82  }
83  }
84  m_controller->update(); // Controllerの状態を更新
85  screen()->redraw(); // スクリーンの再描画
86  }
87  }
88 };
