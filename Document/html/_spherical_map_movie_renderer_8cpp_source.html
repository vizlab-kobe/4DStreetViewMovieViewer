57 void SphericalMapMovieRenderer::exec( kvs::ObjectBase* object, kvs::Camera* camera, kvs::Light* light )
58 {
59  kvs::IgnoreUnusedVariable( light );
60 
61  BaseClass::startTimer();
62  lib4dsv::MovieObject* video = reinterpret_cast<lib4dsv::MovieObject*>( object );
63 
64  if ( !m_texture.isCreated() )
65  {
66  this->create_texture( video ); // textureを作成
67  }
68 
69  if ( !m_shader_program.isCreated() )
70  {
71  this->create_shader_program(); // shader programを作成
72  }
73 
74 // if ( m_type == Centering )
75 // {
76 // this->center_alignment( camera->windowWidth(), camera->windowHeight() );
77 // }
78 
79 // kvs::OpenGL::SetClearDepth( 1.0 );
80  kvs::OpenGL::Clear( GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT ); // カラー/デプスバッファのクリア
81 
82  kvs::OpenGL::WithPushedAttrib p( GL_ALL_ATTRIB_BITS ); // glPushAttrib( GL_ALL_ATTRIB_BITS )
83  {
84  kvs::OpenGL::Disable( GL_DEPTH_TEST ); // glDisable( GL_DEPTH_TEST )：深部テストの無効化
85  kvs::OpenGL::Disable( GL_TEXTURE_1D ); // 1次元テクスチャを無効化
86  kvs::OpenGL::Disable( GL_TEXTURE_3D ); // 3次元テクスチャを無効化
87  kvs::OpenGL::Enable( GL_TEXTURE_2D ); // 2次元テクスチャを有効化
88 
89  kvs::ProgramObject::Binder shader( m_shader_program ); // 描画に使用するProgramObject(m_shader_program)を設定
90  kvs::Texture::Binder unit( m_texture, 0 ); // 使用するテクスチャユニットをm_textureに関連づける
91 
92  const IplImage* frame = video->device().queryFrame(); // cvQueryFrame：fileから1つのフレームを取り出す
93  const int width = frame->width; // フレームの幅
94  const int height = frame->height; // フレームの高さ
95  const char* data = frame->imageData; // BGRBGRBGR... ：画像データへのポインタ
96  m_texture.load( width, height, data ); // GPUへテクスチャデータをダウンロードする
97 
98  m_shader_program.setUniform( "spherical_map", 0 ); // "spherical_map"というデータをテクスチャID:0としてシェーダに渡す
99  m_shader_program.setUniform( "R", object->xform().rotation() ); // "R"というuniform変数に回転行列を指定する
100  m_shader_program.setUniform( "image_size", kvs::Vec2( video->width(), video->height() ) ); // "image_size"というuniform変数に画像の幅/高さを指定する
101  m_shader_program.setUniform( "screen_size", kvs::Vec2( camera->windowWidth(), camera->windowHeight() ) ); // "screen_size"というuniform変数にwindowの幅/高さを指定する
102 
103  kvs::OpenGL::WithPushedMatrix p1( GL_MODELVIEW ); // モデルビューモードに変換（以降モデルビュー変換行列演算を行う）
104  p1.loadIdentity(); // 変換行列を単位行列に初期化する
105  {
106  kvs::OpenGL::WithPushedMatrix p2( GL_PROJECTION ); // 投影変換モードに変換（以降投影変換行列演算を行う）
107  p2.loadIdentity(); // 変換行列を単位行列に初期化する
108  {
109  kvs::OpenGL::SetOrtho( m_left, m_right, m_bottom, m_top, -1, 1 ); // m_left, m_right, m_bottom, m_top, front(-1), back(1)で囲まれる範囲を平行投影する
110  kvs::OpenGL::Begin( GL_QUADS ); // 4つの頂点をペアとしてそれぞれ独立した４角形を描画する
111  kvs::OpenGL::TexCoordVertex( kvs::Vec2( 0.0, 0.0 ), kvs::Vec2( 0.0, 1.0 ) ); // テクスチャ切出し座標と貼付け座標の指定
112  kvs::OpenGL::TexCoordVertex( kvs::Vec2( 0.0, 1.0 ), kvs::Vec2( 0.0, 0.0 ) );
113  kvs::OpenGL::TexCoordVertex( kvs::Vec2( 1.0, 1.0 ), kvs::Vec2( 1.0, 0.0 ) );
114  kvs::OpenGL::TexCoordVertex( kvs::Vec2( 1.0, 0.0 ), kvs::Vec2( 1.0, 1.0 ) );
115  kvs::OpenGL::End(); // 描画終了
116  }
117  }
118  }
119 
120  if ( m_enable_auto_play ) // AutoPlay実行中
121  {
122  m_current_frame_index = m_frame_index; // 現在描画中のframe_indexを取得(以降のm_frame_indexは次の描画への準備)
123 
124  if ( m_enable_reverse_play ) // ReversePlay実行中
125  {
126  int reverse_play_index = video->device().nextFrameIndex() - 2;
127  m_frame_index--;
128 
129  if ( m_enable_loop_play ) // LoopPlay実行中
130  {
131  if ( video->device().nextFrameIndex() == 1 ) // nextFrameIndexが1の時（ 現在描画中のframeIndexは0 ）
132  {
133  reverse_play_index = video->device().numberOfFrames() - 1;
134  m_frame_index = video->device().numberOfFrames() - 1; // nextFrameIndexを一番最後のフレームに指定する
135  }
136  }
137  else // LoopPlay OFFの場合
138  {
139  if ( video->device().nextFrameIndex() == 1 )
140  {
141  reverse_play_index = 0;
142  m_frame_index = 0;
143  }
144  }
145  video->device().setNextFrameIndex( reverse_play_index ); // nextFrameIndexをOpenCVに反映する
146  }
147  else // ReversePlay OFFの場合
148  {
149  m_frame_index++;
150  if ( m_enable_loop_play ) // LooPlay 実行中
151  {
152  if ( video->device().nextFrameIndex() == video->device().numberOfFrames() ) // 最後のフレームまで行ったら
153  {
154  video->device().setNextFrameIndex(0); // 次回は最初のフレームへ
155  m_frame_index = 0;
156  }
157  }
158  else // LoopPlay OFFの場合
159  {
160  if ( video->device().nextFrameIndex() == video->device().numberOfFrames() ) // 最後のフレームまできたら
161  {
162  video->device().setNextFrameIndex( video->device().numberOfFrames() - 1 ); // 以降も最後のフレームを保持
163  m_frame_index = video->device().numberOfFrames() - 1;
164  }
165  }
166  }
167  }
168  else // AutoPlay OFFの場合
169  {
170  const int index = kvs::Math::Clamp( m_frame_index, 0, (int)video->device().numberOfFrames() - 1 );
171  video->device().setNextFrameIndex( index );
172  }
173  BaseClass::stopTimer();
174 }

194 void SphericalMapMovieRenderer::create_texture( const lib4dsv::MovieObject* movie )
195 {
196  const double width = movie->width();
197  const double height = movie->height();
198  m_initial_aspect_ratio = width / height;
199  m_left = 0.0;
200  m_right = 1.0;
201  m_bottom = 0.0;
202  m_top = 1.0;
203 
204  if ( movie->type() == lib4dsv::MovieObject::Gray8 )
205  {
206  m_texture.setPixelFormat( GL_INTENSITY8, GL_LUMINANCE, GL_UNSIGNED_BYTE ); // Set the pixcel format.
207  }
208  else if ( movie->type() == lib4dsv::MovieObject::Color24 )
209  {
210 #ifdef GL_BGR_EXT
211  m_texture.setPixelFormat( GL_RGB8, GL_BGR_EXT, GL_UNSIGNED_BYTE );
212 #else
213  m_texture.setPixelFormat( GL_RGB8, GL_RGB, GL_UNSIGNED_BYTE );
214 #endif
215  }
216  else
217  {
218  kvsMessageError("Unknown pixel color type.");
219  }
220 
221 // const IplImage* frame = movie->device().queryFrame();
222 
223  kvs::Texture::SetEnv( GL_TEXTURE_ENV_MODE, GL_REPLACE ); // textureの合成環境の設定(背景の色を無視してtextureの色に変換)
224  m_texture.setWrapS( GL_REPEAT ); // wrap method for s-axis
225  m_texture.setWrapT( GL_REPEAT ); // wrap method for t-axis
226  m_texture.setMagFilter( GL_LINEAR ); // magnification filter (拡大)
227  m_texture.setMinFilter( GL_LINEAR ); // minification filter (縮小)
228 // m_texture.create( frame->width, frame->height );
229  m_texture.create( width, height ); // texture objectを作成し、これに画像を設定する。
230 }

234  * @brief Calculates centering parameters.
235  * @param width [in] image width
236  * @param height [in] image height
237  */
238 /*==========================================================================*/
239 void SphericalMapMovieRenderer::center_alignment( const double width, const double height )
240 {
241  const double current_aspect_ratio = width / height; // アスペクト比
242  const double aspect_ratio = current_aspect_ratio / m_initial_aspect_ratio;
243  if( aspect_ratio >= 1.0 )
244  {
245  m_left = ( 1.0 - aspect_ratio ) * 0.5;
246  m_right = ( 1.0 + aspect_ratio ) * 0.5;
247  m_bottom = 0.0;
248  m_top = 1.0;
249  }
250  else
251  {
252  m_left = 0.0;
253  m_right = 1.0;
254  m_bottom = ( 1.0 - 1.0 / aspect_ratio ) * 0.5;
255  m_top = ( 1.0 + 1.0 / aspect_ratio ) * 0.5;
256  }
257 }
